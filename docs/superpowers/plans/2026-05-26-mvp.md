# 中控台 MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first Windows-oriented desktop MVP where an operator pastes style numbers, the tool reads local hand-card Excel files, extracts product data/images/SKU/size charts, optionally runs AI generation, and exports a final `.xlsx`.

**Architecture:** Use Electron for local filesystem access and a React renderer for the operator UI. Keep business logic in focused TypeScript modules under `src/main/services` so parsing, image extraction, AI generation, and export can be tested without the UI. The first MVP stops before Doudian product creation.

**Tech Stack:** Electron, Vite, React, TypeScript, Vitest, ExcelJS, fast-xml-parser, OpenAI SDK for OpenRouter, lucide-react, Tailwind CSS.

---

## Scope

This MVP implements the local data-preparation loop only:

1. Paste style numbers.
2. Select one or more hand-card Excel workbooks.
3. Parse product blocks from `手卡资料`.
4. Extract embedded product images as `款号.png`.
5. Parse SKU and size chart data, including suggested weight per size.
6. Export one final `.xlsx` with `商品总表`, `SKU明细`, `尺码表`, and `异常记录`.
7. Optionally run AI generation through OpenRouter after the local pipeline is stable.

This MVP does not log in to Doudian, create product links, or submit products for listing.

```ts
export function parseStyleNumbers(input: string): string[] {
  const seen = new Set<string>();
  const normalized = input
    .replace(/[，、；;]/g, '\n')
    .split(/[\s,\n\r\t]+/g)
    .map(token => token.trim())
    .filter(Boolean);

  const result: string[] = [];
  for (const token of normalized) {
    if (seen.has(token)) continue;
    seen.add(token);
    result.push(token);
  }
  return result;
}
```

`src/main/services/handcardParser.ts`:

```ts
import path from 'node:path';
import ExcelJS from 'exceljs';
import type { ProductRecord, SizeChartRow, SkuItem } from '../types';

export interface HandcardParseResult {
  products: ProductRecord[];
  duplicateStyleNumbers: string[];
}

const SHEET_NAME = '手卡资料';
const SIZE_PATTERN = /(5XL|4XL|3XL|2XL|XL|L|M|S)$/;

function cellText(row: ExcelJS.Row, column: number): string {
  const value = row.getCell(column).value;
  if (value == null) return '';
  if (value instanceof Date) return value.toISOString().slice(0, 10);
  if (typeof value === 'object' && 'text' in value) return String(value.text).trim();
  return String(value).trim();
}

function toNumber(value: string): number | null {
  if (!value) return null;
  const parsed = Number(value);
  return Number.isFinite(parsed) ? parsed : null;
}

function splitColors(value: string): string[] {
  return value.split(/[\/、,，\s]+/g).map(color => color.trim()).filter(Boolean);
}

function splitSpec(spec: string): { color: string; size: string } {
  const match = spec.match(SIZE_PATTERN);
  if (!match) return { color: spec, size: '' };
  return { color: spec.slice(0, match.index).trim(), size: match[1] };
}

function parseSkuItems(sheet: ExcelJS.Worksheet, startRow: number, nextStartRow: number, styleNumber: string, finalStorePrice: number | null): SkuItem[] {
  const items: SkuItem[] = [];
  for (let rowNumber = startRow; rowNumber < Math.min(nextStartRow, startRow + 24); rowNumber += 1) {
    const row = sheet.getRow(rowNumber);
    for (let col = 2; col <= 14; col += 2) {
      const spec = cellText(row, col);
      const merchantCode = cellText(row, col + 1);
      if (!spec || spec === '颜色规格' || !merchantCode) continue;
      const { color, size } = splitSpec(spec);
      items.push({ styleNumber, color, size, merchantCode, finalStorePrice, suggestedWeight: '', note: '' });
    }
  }
  return items;
}

function parseSizeRows(sheet: ExcelJS.Worksheet, startRow: number, styleNumber: string): SizeChartRow[] {
  const headerRow = sheet.getRow(startRow);
  const sizeColumns: Array<{ size: string; col: number; bodyType: string }> = [];
  for (let col = 20; col <= 34; col += 1) {
    const size = cellText(headerRow, col);
    if (/^(L|XL|2XL|3XL|4XL|5XL)$/.test(size)) {
      sizeColumns.push({ size, col, bodyType: cellText(sheet.getRow(startRow + 1), col) });
    }
  }

  const rowsBySize = new Map<string, SizeChartRow>();
  for (const item of sizeColumns) {
    rowsBySize.set(item.size, { styleNumber, size: item.size, bodyType: item.bodyType, suggestedWeight: '', measurements: {} });
  }

  for (let rowNumber = startRow + 2; rowNumber < startRow + 14; rowNumber += 1) {
    const row = sheet.getRow(rowNumber);
    const label = cellText(row, 22);
    if (!label) continue;
    for (const item of sizeColumns) {
      const value = cellText(row, item.col);
      if (!value) continue;
      const target = rowsBySize.get(item.size);
      if (!target) continue;
      if (label === '建议体重') target.suggestedWeight = value;
      else target.measurements[label] = value;
    }
  }

  return Array.from(rowsBySize.values());
}

function applySuggestedWeights(skuItems: SkuItem[], sizeRows: SizeChartRow[]) {
  const bySize = new Map(sizeRows.map(row => [row.size, row.suggestedWeight]));
  for (const sku of skuItems) {
    sku.suggestedWeight = bySize.get(sku.size) || '';
    sku.note = sku.suggestedWeight ? `建议体重${sku.suggestedWeight}斤` : '';
  }
}

export async function parseHandcardWorkbook(filePath: string): Promise<HandcardParseResult> {
  const workbook = new ExcelJS.Workbook();
  await workbook.xlsx.readFile(filePath);
  const sheet = workbook.getWorksheet(SHEET_NAME);
  if (!sheet) return { products: [], duplicateStyleNumbers: [] };

  const dataRows: number[] = [];
  sheet.eachRow((row, rowNumber) => {
    if (cellText(row, 4) === '款号') {
      const next = sheet.getRow(rowNumber + 1);
      if (cellText(next, 4)) dataRows.push(rowNumber + 1);
    }
  });

  const products: ProductRecord[] = [];
  const seen = new Set<string>();
  const duplicateStyleNumbers = new Set<string>();

  for (let index = 0; index < dataRows.length; index += 1) {
    const rowNumber = dataRows[index];
    const nextStartRow = dataRows[index + 1] || sheet.rowCount + 1;
    const row = sheet.getRow(rowNumber);
    const styleNumber = cellText(row, 4);
    if (seen.has(styleNumber)) duplicateStyleNumbers.add(styleNumber);
    seen.add(styleNumber);

    const finalStorePrice = toNumber(cellText(row, 13));
    const skuItems = parseSkuItems(sheet, rowNumber + 11, nextStartRow, styleNumber, finalStorePrice);
    const sizeRows = parseSizeRows(sheet, rowNumber, styleNumber);
    applySuggestedWeights(skuItems, sizeRows);

    products.push({
      styleNumber,
      supplier: cellText(row, 3),
      brand: cellText(row, 7),
      systemCategory: cellText(row, 6),
      colors: splitColors(cellText(row, 5)),
      originalName: cellText(row, 16),
      taxPrice: toNumber(cellText(row, 10)),
      companyPrice: toNumber(cellText(row, 11)),
      finalStorePrice,
      fabricName: cellText(sheet.getRow(rowNumber + 1), 16),
      composition: [cellText(sheet.getRow(rowNumber + 3), 3), cellText(sheet.getRow(rowNumber + 3), 8)].filter(Boolean).join('；'),
      sourceFile: path.resolve(filePath),
      sourceSheet: SHEET_NAME,
      sourceRow: rowNumber,
      skuItems,
      sizeRows,
      warnings: []
    });
  }

  return { products, duplicateStyleNumbers: Array.from(duplicateStyleNumbers) };
}
```

```ts
  sourceRow: number;
}

export async function extractProductImages(workbookPath: string, outputDir: string, requests: ImageAnchorRequest[]): Promise<Map<string, string>> {
  await fs.mkdir(outputDir, { recursive: true });
  const workbook = new ExcelJS.Workbook();
  await workbook.xlsx.readFile(workbookPath);
  const sheet = workbook.getWorksheet('手卡资料');
  const result = new Map<string, string>();
  if (!sheet) return result;

  const requestsByRow = new Map(requests.map(request => [request.sourceRow, request.styleNumber]));
  const mediaById = new Map(workbook.model.media.map(media => [media.index, media]));

  for (const image of sheet.getImages()) {
    const row = image.range.tl.nativeRow + 1;
    const styleNumber = requestsByRow.get(row);
    if (!styleNumber) continue;
    const media = mediaById.get(image.imageId);
    if (!media?.buffer) continue;

    const outputPath = path.join(outputDir, `${styleNumber}.png`);
    await fs.writeFile(outputPath, media.buffer);
    result.set(styleNumber, outputPath);
  }

  return result;
}
```

```ts
import path from 'node:path';
import { parseStyleNumbers } from './styleNumber';
import { parseHandcardWorkbook } from './handcardParser';
import { extractProductImages } from './workbookImages';
import { exportBatchWorkbook } from './exportWorkbook';
import type { BatchRow, ProductRecord } from '../types';

export interface ProcessBatchWithoutAiInput {
  styleNumberText: string;
  workbookPaths: string[];
  outputDir: string;
}

export interface ProcessBatchWithoutAiResult {
  rows: BatchRow[];
  workbookPath: string;
}

export async function processBatchWithoutAi(input: ProcessBatchWithoutAiInput): Promise<ProcessBatchWithoutAiResult> {
  const requested = parseStyleNumbers(input.styleNumberText);
  const productByStyle = new Map<string, ProductRecord>();

  for (const workbookPath of input.workbookPaths) {
    const parsed = await parseHandcardWorkbook(workbookPath);
    for (const product of parsed.products) {
      if (!productByStyle.has(product.styleNumber)) productByStyle.set(product.styleNumber, product);
    }

    const matchesInWorkbook = requested
      .map(styleNumber => productByStyle.get(styleNumber))
      .filter((product): product is ProductRecord => Boolean(product))
      .filter(product => product.sourceFile === path.resolve(workbookPath));

    const images = await extractProductImages(workbookPath, path.join(input.outputDir, 'images'), matchesInWorkbook.map(product => ({ styleNumber: product.styleNumber, sourceRow: product.sourceRow })));
    for (const product of matchesInWorkbook) {
      product.imagePath = images.get(product.styleNumber);
      if (!product.imagePath) product.warnings.push('缺少产品图');
    }
  }

  const products: ProductRecord[] = [];
  const errors: Array<{ styleNumber: string; reason: string }> = [];
  const rows: BatchRow[] = requested.map(styleNumber => {
    const product = productByStyle.get(styleNumber);
    if (!product) {
      errors.push({ styleNumber, reason: '款号未找到' });
      return { styleNumber, status: 'error', reason: '款号未找到' };
    }
    products.push(product);
    return { styleNumber, status: product.warnings.length ? 'warning' : 'success', reason: product.warnings.join('；'), product };
  });

  const workbookPath = path.join(input.outputDir, `中控台导出_${new Date().toISOString().slice(0, 10)}.xlsx`);
  await exportBatchWorkbook({ outputPath: workbookPath, products, errors });
  return { rows, workbookPath };
}
```
